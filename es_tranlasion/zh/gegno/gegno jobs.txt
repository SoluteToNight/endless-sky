mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "收集10吨铝"
	description "开采并收集 10 吨铝，然后将矿物带回 <origin> 的杰格诺。"
	source "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
	to accept
		"cargo space" > 10
	on visit
		dialog "您已返回 <origin>，但货舱中没有 10 吨铝。确保在着陆前开采 10 吨铝并将其收集到您的货物中。"
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno mining payment dialog"
